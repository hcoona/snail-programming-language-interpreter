TOKENS: ID INT

program -> stmt_list

stmt_list ->
  | stmt_list stmt
  | stmt

stmt ->
  | simple_stmt ';'
  | EPSILON

simple_stmt ->
  | assign_stmt
  | print_stmt

assign_stmt -> ID '=' expr

print_stmt ->
  | PRINT expr

expr ->
  | '(' expr ')'
  | expr '+' expr
  | expr '*' expr
  | '-' expr
  | INT
  | ID
